47 inline const char* getMime(const std::string& path)
48 {
49 const char* p = path.c_str();
50 const char* ext = strrchr(p, '.');
51 if (!ext || *(ext + 1) == '\0')
52 return "application/octet-stream";
53
54 ext++; // skip '.'
55
56 if (!strcasecmp(ext, "html") || !strcasecmp(ext, "htm")) return "text/html";
57 if (!strcasecmp(ext, "css")) return "text/css";
58 if (!strcasecmp(ext, "js")) return "application/javascript";
59 if (!strcasecmp(ext, "json")) return "application/json";
60 if (!strcasecmp(ext, "txt")) return "text/plain";
61 if (!strcasecmp(ext, "xml")) return "application/xml";
62
63 if (!strcasecmp(ext, "png")) return "image/png";
64 if (!strcasecmp(ext, "jpg") || !strcasecmp(ext, "jpeg")) return "image/jpeg";
65 if (!strcasecmp(ext, "svg")) return "image/svg+xml";
66 if (!strcasecmp(ext, "ico")) return "image/x-icon";
67 if (!strcasecmp(ext, "pdf")) return "application/pdf";
68
69 return "application/octet-stream";
70 }
